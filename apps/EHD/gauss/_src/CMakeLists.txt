# CMake input file of the SWIG wrapper around "gauss.so"

INCLUDE(${SWIG_USE_FILE})

INCLUDE_DIRECTORIES(${PYTHON_INCLUDE_PATH})

FILE(GLOB SRCS *.h *.cpp *.inl *.swg)
FILE(GLOB ISRCS *.i)

SET(CMAKE_SWIG_FLAGS "")
SET_SOURCE_FILES_PROPERTIES(${ISRCS} PROPERTIES CPLUSPLUS ON)

SET(SWINCFLAGS 
-I${PROJECT_SOURCE_DIR}/gauss/src
)
SET_SOURCE_FILES_PROPERTIES(${ISRCS} PROPERTIES SWIG_FLAGS "${SWINCFLAGS}")


IF (CMAKE_VERSION VERSION_LESS 3.8.0)
    SWIG_ADD_MODULE(gaussi python ${ISRCS} ${SRCS})
ELSE ()
    SWIG_ADD_LIBRARY(gaussi LANGUAGE python SOURCES ${ISRCS} ${SRCS})
ENDIF ()
MACRO_DebugPostfix(_gaussi)

# ---

INCLUDE_DIRECTORIES( ${PROJECT_SOURCE_DIR}/gauss/src
)

SWIG_LINK_LIBRARIES(gaussi 
                    gauss ${PYTHON_LIBRARIES}
)
