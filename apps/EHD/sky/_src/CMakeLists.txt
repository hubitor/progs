# CMake input file of the SWIG wrapper around "sky.so"

INCLUDE(${SWIG_USE_FILE})

INCLUDE_DIRECTORIES(${PYTHON_INCLUDE_PATH})

FILE(GLOB SRCS *.h *.cpp *.inl *.swg)
FILE(GLOB ISRCS *.i)

SET(CMAKE_SWIG_FLAGS "")
SET_SOURCE_FILES_PROPERTIES(${ISRCS} PROPERTIES CPLUSPLUS ON)

SET(SWINCFLAGS 
-I${PROJECT_SOURCE_DIR}/sky/src
)
SET_SOURCE_FILES_PROPERTIES(${ISRCS} PROPERTIES SWIG_FLAGS "${SWINCFLAGS}")



SWIG_ADD_MODULE(skyi python ${ISRCS} ${SRCS})
MACRO_DebugPostfix(_skyi)

# ---

INCLUDE_DIRECTORIES( ${PROJECT_SOURCE_DIR}/sky/src
)

SWIG_LINK_LIBRARIES(skyi 
                    sky ${PYTHON_LIBRARIES}
)
